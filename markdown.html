def convert_markdown_2_html(inputfile):
    with open(inputfile, 'r') as f:
        text = f.read()
        html = markdown.markdown(text, extensions=[FencedCodeExtension()])

    temp = inputfile.strip('.')
    outputfile = temp[0] + '.html'

    with open(outputfile, 'w') as f:
        f.write(html)
